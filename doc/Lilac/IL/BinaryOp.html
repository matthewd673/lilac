Overview

A BinaryOp is an Expression which computes a value from two operands. The two operands must have the same type.

def calculate
  # calculations can only be performed on constants
  if !@left.is_a?(Constant) || !@right.is_a?(Constant)
    return nil
  end

  left = @left
  right = @right

  case @op
  when Operator::ADD
    left.value + right.value
  when Operator::SUB
    left.value - right.value
  when Operator::MUL
    left.value * right.value
  when Operator::DIV
    left.value / right.value
  when Operator::EQ
    left.value == right.value ? 1 : 0
  when Operator::NEQ
    left.value != right.value ? 1 : 0
  when Operator::LT
    left.value < right.value ? 1 : 0
  when Operator::GT
    left.value > right.value ? 1 : 0
  when Operator::LEQ
    left.value <= right.value ? 1 : 0
  when Operator::GEQ
    left.value >= right.value ? 1 : 0
  when Operator::OR
    left.value != 0 || right.value != 0 ? 1 : 0
  when Operator::AND
    left.value != 0 && right.value != 0 ? 1 : 0
  else T.absurd(self)
  end
end

def eql?(other)
  if other.class != BinaryOp
    return false
  end

  other = T.cast(other, BinaryOp)

  op.eql?(other.op) and left.eql?(other.left) and right.eql?(other.right)
end
