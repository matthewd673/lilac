def to_s
  case self
  when Void then "void"
  when U8 then "u8"
  when U16 then "u16"
  when U32 then "u32"
  when U64 then "u64"
  when I8 then "i8"
  when I16 then "i16"
  when I32 then "i32"
  when I64 then "i64"
  when F32 then "f32"
  when F64 then "f64"
  else
    T.absurd(self)
  end
end
