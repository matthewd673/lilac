def get_idom(block)
  incoming = @incoming[block]
  if not incoming
    nil
  else
    # there is only one IDOM in a valid graph
    idom_edge = incoming.to_a[0]
    if not idom_edge
      return nil
    end

    return idom_edge.from
  end
end

def get_sdom(block)
  sdom = T.let(Set[], T::Set[BB])
  incoming = @incoming[block]
  if not incoming
    return Set[]
  end

  incoming.each { |e|
    sdom = sdom | get_sdom(e.from)
  }

  return sdom
end
