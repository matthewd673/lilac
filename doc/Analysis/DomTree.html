def dom_by(node, &block)

    dom_by(o.to, &block)
  }

def idom(node)
  incoming = @incoming[node]
  if not incoming
    nil
  else
    # there is only one IDOM in a valid graph
    idom_edge = incoming.to_a[0]
    if not idom_edge
      return nil
    end

    return idom_edge.from
  end
end

def sdom(node, &block)
  # yield each predecessor and recurse on them
  each_incoming(node) { |i|
    yield i.from
    sdom(node, &block)
  }
end
