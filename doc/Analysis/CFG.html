def add_node(node)
  @nodes.push(node)
  if node.entry and node.entry.is_a?(IL::Label)
    @label_map[T.unsafe(node.entry).name] = node
  end
end

def max_block_id
  max = -1 # also = to ENTRY so be careful
  each_block { |b|
    if b.id > max then max = b.id end
  }
  return max
end
