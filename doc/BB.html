def self.create_blocks(program)
  blocks = [Block.new]

  program.each_stmt { |s|
    # mark beginning of a block
    if s.is_a?(IL::Label) and not blocks[-1].empty?
      blocks.push(Block.new)
    end

    blocks[-1].push_stmt(s)

    # mark end of a block
    if s.is_a?(IL::Jump) # block will never be empty due to above push
      blocks.push(Block.new)
    end
  }

  return blocks
end
