def eql?(other)
  if not other.class == Program
    return false
  end

  other = T.cast(other, Program)

  stmt_list.eql?(other.stmt_list) and func_map.eql?(other.func_map)
end

def to_s
  str = ""
  @stmt_list.each { |i|
    str += i.to_s + "\n"
  }
  return str
end
