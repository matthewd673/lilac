def to_s
  param_str = ""
  @params.each { |p|
    param_str += "#{p}, "
  }
  param_str.chomp!(", ")

  stmt_str = ""
  @stmt_list.each { |s|
    stmt_str += "#{s}\n"
  }

  return "func #{@name}(#{param_str}) -> #{@ret_type}:\n#{stmt_str}\nend"
end
