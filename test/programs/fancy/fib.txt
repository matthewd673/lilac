extern func console log(i32) -> void

func fib(i32 n@0) -> i32
  i32 %0 = n@0 == 0i32
  jz %0 L0
  ret 0i32
  jmp L1
  L0:
  L1:
  i32 %1 = n@0 == 1i32
  jz %1 L2
  ret 1i32
  jmp L3
  L2:
  L3:
  i32 %2 = n@0 - 1i32
  i32 %3 = call fib(%2)
  i32 a@1 = %3
  i32 %4 = n@0 - 2i32
  i32 %5 = call fib(%4)
  i32 b@1 = a@1
  i32 %6 = a@1 + b@1
  ret %6
end

func main() -> void

  i32 ans@1 = %7
  void extern call console log(ans@1)
  ret void
end
